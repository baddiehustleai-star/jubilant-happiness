rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Users can read/write their own user document
    match /users/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
    
    // Users can read/write their own upload history
    match /users/{userId}/uploads/{uploadId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
    
    // Users can read/write their own usage stats
    match /users/{userId}/usage/{usageId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
    
    // Public read access to subscription plans (for pricing page)
    match /plans/{planId} {
      allow read: if true;
      allow write: if false; // Only admin can modify plans
    }
    
    // Deny all other access
    match /{document=**} {
      allow read, write: if false;
    }
  }
}

// Storage Rules
service firebase.storage {
  match /b/{bucket}/o {
    // Users can upload to their own folder
    match /users/{userId}/{allPaths=**} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
    
    // Public read access to processed images (for sharing)
    match /processed/{allPaths=**} {
      allow read: if true;
      allow write: if false;
    }
  }
}
